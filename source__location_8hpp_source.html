78  source_location loc;
79  loc.file_ = file;
80  loc.func_ = func;
81  loc.line_ = line;
82  loc.column_ = column;
83  try {
84  // get the current MPI rank iff the MPI environment is active
85  int is_initialized, is_finalized;
86  MPI_Initialized(&is_initialized);
87  MPI_Finalized(&is_finalized);
88  if (static_cast<bool>(is_initialized) && !static_cast<bool>(is_finalized)) {
89  int rank;
90  int err = MPI_Comm_rank(MPI_COMM_WORLD, &rank);
91  if (err == MPI_SUCCESS) {
92  loc.rank_ = std::make_optional(rank);
93  }
94  }
95  } catch (...) {
96  // something went wrong during the MPI calls -> no information could be retrieved
97  loc.rank_ = std::nullopt;
98  }
99  return loc;

137 
138  // no stack addresses could be retrieved

141  }
142

144  char** symbollist = backtrace_symbols(addrlist.data(), addrlen);
145  for (int i = 0; i < addrlen; ++i) {
146  symbols.emplace_back(symbollist[i]);
147  }
148  free(symbollist);
149  }
150 
151  // iterate over the returned symbol lines -> skip the first and second symbol because they are unimportant
152  for (std::size_t i = 2; i < symbols.size(); ++i) {
153  fmt::format_to(buf, " #{:<6}", symbols.size() - i);
154 
155  // file_name(function_name+offset) -> split the symbol line accordingly
156  const std::size_t position1 = std::min(symbols[i].find_first_of("("), symbols[i].size());
157  const std::size_t position2 = std::min(symbols[i].find_first_of("+"), symbols[i].size());
158  const std::size_t position3 = std::min(symbols[i].find_first_of(")"), symbols[i].size());
159

161  std::string_view file_name = symbols_view.substr(0, position1);
162  std::string function_name = symbols[i].substr(position1 + 1, position2 - position1 - 1);
163  std::string_view function_offset = symbols_view.substr(position2, position3 - position2 - 1);
164

166  if (!file_name.empty() && !function_name.empty() && !function_offset.empty()) {

170

174  } else {

176  fmt::format_to(buf, "{}: {}() [{}]\n", file_name, function_name, function_offset);
177  }

179  } else {
