399  /*
400  * @brief Spawns a number of MPI processes according to the previously set options.
401  * @details Removes empty values before getting passed to *MPI_COMM_SPAWN*.
402  * @tparam return_type either @ref mpicxx::spawn_result or @ref mpicxx::spawn_result_with_errcodes
403  * @return the result of the spawn invocation
404  *
405  * @pre The executable name **must not** be empty.
406  * @pre All command line arguments **must not** be empty.
407  * @pre maxprocs **must not** be less or equal than `0` or greater than the maximum possible number of processes
408  * (@ref mpicxx::universe_size()).
409  * @pre root **must not** be less than `0` and greater or equal than the size of the communicator (set via
410  * @ref set_communicator(MPI_Comm) or default *MPI_COMM_WORLD*).
411  * @pre comm **must not** be *MPI_COMM_NULL*.
412  *
413  * @assert_precondition{ If the executable name is empty. \n
414  * If any command line argument is empty. \n
415  * If maxprocs is invalid. \n
416  * If root isn't a legal root. \n
417  * If comm is the null communicator (*MPI_COMM_NULL*). }
418  *
419  * @calls{
420  * int MPI_Comm_spawn(const char *command, char *argv[], int maxprocs, MPI_Info info, int root, MPI_Comm comm, MPI_Comm *intercomm, int array_of_errcodes[]); // exactly once
421  * }
422  */
423  template <typename return_type>
424  return_type spawn_impl() {
425  MPICXX_ASSERT_PRECONDITION(this->legal_command(command_), "Attempt to use the executable name which is only an empty string!");
426  MPICXX_ASSERT_PRECONDITION(this->legal_argv(argvs_).first,
427  "Attempt to use the {}-th command line argument which is only an empty string!", this->legal_argv(argvs_).second);
428  MPICXX_ASSERT_PRECONDITION(this->legal_maxprocs(maxprocs_),
429  "Attempt to use the maxprocs value (which is {}), which falls outside the valid range (0, {}]!",
430  maxprocs_, mpicxx::universe_size().value_or(std::numeric_limits<int>::max()));
431  MPICXX_ASSERT_PRECONDITION(this->legal_root(root_, comm_),
432  "The previously set root '{}' isn't a valid root in the current communicator!", root_);
433  MPICXX_ASSERT_PRECONDITION(this->legal_communicator(comm_), "Can't use the null communicator!");
434 
435  return_type res(maxprocs_);
436 
437  // determine whether the placeholder MPI_ERRCODES_IGNORE shall be used or a "real" std::vector
438  auto errcode = [&res]() {
439  if constexpr (std::is_same_v<return_type, spawn_result_with_errcodes>) {
440  return res.errcodes_.data();
441  } else {
442  return MPI_ERRCODES_IGNORE;
443  }
444  }();
445 
446  if (argvs_.empty()) {
447  // no additional arguments provided -> use MPI_ARGV_NULL
448  MPI_Comm_spawn(command_.c_str(), MPI_ARGV_NULL, maxprocs_, info_.get(),
449  root_, comm_, &res.intercomm_, errcode);
450  } else {
451  // convert additional arguments to char**
452  std::vector<char*> argvs_ptr;
453  argvs_ptr.reserve(argvs_.size() + 1);
454  for (auto& str : argvs_) {
455  argvs_ptr.emplace_back(str.data());
456  }
457  // add null termination
458  argvs_ptr.emplace_back(nullptr);
459 
460  MPI_Comm_spawn(command_.c_str(), argvs_ptr.data(), maxprocs_, info_.get(),
461  root_, comm_, &res.intercomm_, errcode);
462  }
463  return res;
464  }
